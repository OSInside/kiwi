156    def _process_install_requests_bootstrap_prebuild_root(
157        self, bootstrap_package: str
158    ) -> CommandCallT:
159        """
160        Process bootstrap phase (no chroot) using a prebuilt bootstrap
161        package. The package has to provide a tarball below the
162        directory /var/cache/kiwi/bootstrap/PACKAGE_NAME.ARCH.tar.xz
163        and will be unpacked to serve as the bootstrap root.
164
165        The optionally listed packages in the kiwi bootstrap section
166        will be installed as part of a chroot install and returned
167        as command instance
168
169        :param str bootstrap_package:
170            package name of the bootstrap package containing the
171            bootstrap root as a tarball
172
173        :return: process results as command instance
174
175        :rtype: command_call_type
176        """
177        # Install prebuilt bootstrap package on build host
178        update_command = [
179            'apt-get'
180        ] + self.apt_get_args + self.custom_args + [
181            'update'
182        ]
183        install_command = [
184            'apt-get'
185        ] + self.apt_get_args + self.custom_args + [
186            'install', bootstrap_package
187        ]
188        Command.run(
189            update_command, self.command_env
190        )
191        Command.run(
192            install_command, self.command_env
193        )
194        # Unpack prebuilt bootstrap root tarball as new root
195        bootstrap_root_tarball = '/var/lib/bootstrap/{0}.{1}.tar.xz'.format(
196            bootstrap_package, defaults.PLATFORM_MACHINE
197        )
198        if not os.path.isfile(bootstrap_root_tarball):
199            raise KiwiFileNotFound(
200                f'bootstrap tarball: {bootstrap_root_tarball!r} not found'
201            )
202        Command.run(
203            ['tar', '-C', self.root_dir, '-xf', bootstrap_root_tarball]
204        )
205        # Install eventual bootstrap packages as standard system install
206        return self.process_install_requests()

208    def _process_install_requests_bootstrap(self) -> CommandCallT:
209        """
210        Process package install requests for bootstrap phase (no chroot).
211
212        :raises KiwiDebianBootstrapError
213
214        :return: process results in command type
215
216        :rtype: CommandCallT
217        """
218        # we invoke apt-get install to download all the essential packages.
219        # With DPkg::Pre-Install-Pkgs, we specify a shell command that will
220        # receive the list of packages that will be installed on stdin.
221        # By configuring Debug::pkgDpkgPm=1, apt-get install will not
222        # actually execute any dpkg commands, so all it does is download
223        # the essential debs and tell us their full in the apt cache without
224        # actually installing them.
225        try:
226            # TODO: Drop once apt 2.5.4 is widely available.
227            pathlib.Path(f'{self.root_dir}/var/lib/dpkg').mkdir(
228                parents=True, exist_ok=True
229            )
230            # Older versions of apt look for the status file both within and
231            # without the tree, but do not create the status file within the
232            # tree.
233            pathlib.Path(f'{self.root_dir}/var/lib/dpkg/status.kiwi').touch()
234            # Some versions of apt don't create the status file outside the
235            # tree, e.g. 2.4.14.
236            pathlib.Path('/var/lib/dpkg/status.kiwi').touch()
237            pathlib.Path(f'{self.root_dir}/var/lib/dpkg/available').touch()
238
239            if 'apt' not in self.package_requests:
240                self.package_requests.append('apt')
241            update_cache = [
242                'apt-get'
243            ] + self.apt_get_args + self.custom_args + [
244                'update'
245            ]
246            result = Command.run(
247                update_cache, self.command_env
248            )
249            log.debug(
250                'Apt update: {0} {1}'.format(result.output, result.error)
251            )
252            package_names = Temporary(prefix='kiwi_debs_').new_file()
253            self.bootstrap_packages = [
254                '?essential'
255            ] + self.package_requests
256            download_bootstrap = [
257                'apt-get'
258            ] + self.apt_get_args + self.custom_args + [
259                'install',
260                '-oDebug::pkgDPkgPm=1',
261                f'-oDPkg::Pre-Install-Pkgs::=cat >{package_names.name}'
262            ] + self.bootstrap_packages
263            # Download solved bootstrap packages
264            result = Command.run(
265                download_bootstrap, self.command_env
266            )
267            log.debug(
268                'Apt download: {0} {1}'.format(result.output, result.error)
269            )
270            # Extract bootstrap packages
271            with open(package_names.name) as packages:
272                solved_debootstrap_packages = [p.rstrip() for p in packages]
273            self.command_env['PATH'] = '$PATH:/usr/bin:/bin:/usr/sbin:/sbin'
274            for package in solved_debootstrap_packages:
275                Command.run(
276                    [
277                        'bash', '-c',
278                        'dpkg-deb --fsys-tarfile {0} | tar -C {1} -x'.format(
279                            package, self.root_dir
280                        )
281                    ], self.command_env
282                )
283            # Run package scripts. Unfortuantely Debian based systems
284            # requires special sauce for bootstrap. See the exceptions
285            # we have to apply below:
286            #
287            # * manual order is required to make sure users(root) exists
288            # * the usr-merge strategy only works after bootstrap
289            #
290            # 1. Pass: Run package scripts, manual order
291            self._run_bootstrap_scripts(
292                solved_debootstrap_packages,
293                only_for=['base-passwd'], skip=['usrmerge']
294            )
295            # 2. Pass: Run package scripts in apt order
296            self._run_bootstrap_scripts(
297                solved_debootstrap_packages,
298                skip=['usrmerge']
299            )
300            self.cleanup_requests()
301            return Command.call(
302                update_cache, self.command_env
303            )
304        except Exception as e:
305            raise KiwiDebianBootstrapError(
306                f'{type(e).__name__}: {format(e)}'
307            )

309    def process_install_requests(self) -> CommandCallT:
310        """
311        Process package install requests for image phase (chroot)
312
313        :return: process results in command type
314
315        :rtype: namedtuple
316        """
317        update_command = ['chroot', self.root_dir, 'apt-get']
318        update_command.extend(
319            Path.move_to_root(self.root_dir, self.apt_get_args)
320        )
321        update_command.extend(self.custom_args)
322        update_command.append('update')
323        Command.run(update_command, self.command_env)
324
325        apt_get_command = ['chroot', self.root_dir, 'apt-get']
326        apt_get_command.extend(
327            Path.move_to_root(self.root_dir, self.apt_get_args)
328        )
329        apt_get_command.extend(self.custom_args)
330        apt_get_command.append('install')
331        # Due to the special bootstrap process, the packages unpacked
332        # during bootstrap are not properly listed in the apt index.
333        # Therefore the bootstrap packages are added to the install
334        # phase which causes an install of this packages again to
335        # fix the apt index and provide a consistent system from
336        # an apt perspective.
337        apt_get_command.extend(self.bootstrap_packages)
338        apt_get_command.extend(self._package_requests())
339
340        return Command.call(
341            apt_get_command, self.command_env
342        )

344    def process_delete_requests(self, force: bool = False) -> CommandCallT:
345        """
346        Process package delete requests (chroot)
347
348        :param bool force: force deletion: true|false
349
350        :raises KiwiRequestError: if none of the packages to delete is
351            installed
352        :return: process results in command type
353
354        :rtype: namedtuple
355        """
356        delete_items = []
357        for delete_item in self._package_requests():
358            try:
359                Command.run(
360                    ['chroot', self.root_dir, 'dpkg', '-l', delete_item]
361                )
362                delete_items.append(delete_item)
363            except Exception:
364                # ignore packages which are not installed
365                pass
366        if not delete_items:
367            raise KiwiRequestError(
368                'None of the requested packages to delete are installed'
369            )
370
371        # deleting debs for some reason can also trigger package
372        # scripts from the libc-bin (glibc) package. This often
373        # results in calls like ldconfig which failed unless the
374        # system would be really running. Since the deletion
375        # happens via chroot and with the system in offline mode
376        # many attempts to uninstall a package, even cracefully
377        # failed for this reason. So far I only saw the workaround
378        # to make ldconfig a noop during the process of uninstall.
379        # This is not a nice solution and I dislike it. However,
380        # the only one I could come up with to turn the package
381        # uninstall in a useful feature in kiwi.
382        Command.run(
383            [
384                'cp',
385                f'{self.root_dir}/usr/sbin/ldconfig',
386                f'{self.root_dir}/usr/sbin/ldconfig.orig'
387            ]
388        )
389        Command.run(
390            [
391                'cp',
392                f'{self.root_dir}/usr/bin/true',
393                f'{self.root_dir}/usr/sbin/ldconfig'
394            ]
395        )
396        if force:
397            # force deleting debs only worked well for me when ignoring
398            # the pre/post-inst and pre/post-remove codings. No idea why it
399            # ends in so many conflicts but if you want to force get rid
400            # of stuff this was the only way I could come up with. There
401            # are still cases when it does not work depending on the many
402            # code that runs on deleting
403            for package in delete_items:
404                pre_delete_pattern = \
405                    f'{self.root_dir}/var/lib/dpkg/info/{package}*.pre*'
406                post_delete_pattern = \
407                    f'{self.root_dir}/var/lib/dpkg/info/{package}*.post*'
408                for delete_file in glob.iglob(pre_delete_pattern):
409                    Path.wipe(delete_file)
410                for delete_file in glob.iglob(post_delete_pattern):
411                    Path.wipe(delete_file)
412
413            apt_get_command = ['chroot', self.root_dir, 'dpkg']
414            apt_get_command.extend(
415                [
416                    '--remove',
417                    '--force-remove-reinstreq',
418                    '--force-remove-essential',
419                    '--force-depends'
420                ]
421            )
422            apt_get_command.extend(delete_items)
423        else:
424            apt_get_command = ['chroot', self.root_dir, 'apt-get']
425            apt_get_command.extend(
426                Path.move_to_root(self.root_dir, self.apt_get_args)
427            )
428            apt_get_command.extend(self.custom_args)
429            apt_get_command.extend(['--auto-remove', 'remove'])
430            apt_get_command.extend(delete_items)
431
432        return Command.call(
433            apt_get_command, self.command_env
434        )

487    def match_package_installed(
488        self, package_name: str, package_manager_output: str
489    ) -> bool:
490        """
491        Match expression to indicate a package has been installed
492
493        This match for the package to be installed in the output
494        of the apt-get command is not 100% accurate. There might
495        be false positives due to sub package names starting with
496        the same base package name
497
498        :param str package_name: package_name
499        :param str package_manager_output: apt-get status line

501        :returns: True|False
502
503        :rtype: bool
504        """
505        return bool(
506            re.match(
507                '.*Unpacking {0}.*'.format(re.escape(package_name)),
508                package_manager_output
509            )
510        )

537    def _run_bootstrap_scripts(
538        self, solved_debootstrap_packages: List[str],
539        only_for: List[str] = [], skip: List[str] = []
540    ):
541        # TODO: this should not be needed but without setting
542        # the following environment variables no package pre/post
543        # script completes its task. I leave it up to the Debian
544        # experts to provide a fix if needed.
545        self.command_env['DPKG_MAINTSCRIPT_NAME'] = 'true'
546        self.command_env['DPKG_MAINTSCRIPT_PACKAGE'] = 'libc6'
547
548        post_script_dir = Temporary(
549            prefix='kiwi_debpost.', path=self.root_dir
550        ).new_dir()
551        for package in solved_debootstrap_packages:
552            package_base_name = os.path.basename(package)
553            go_ahead = False if only_for else True
554            for name in only_for:
555                if name in package_base_name:
556                    go_ahead = True
557                    break
558            for name in skip:
559                if name in package_base_name:
560                    go_ahead = False
561                    break
562            if not go_ahead:
563                continue
564            log.debug(
565                f'Running pre/post scripts for: {package_base_name}'
566            )
567            package_metadata_dir = \
568                f'{post_script_dir.name}/{os.path.basename(package)}'
569            Command.run(
570                ['dpkg', '-e', package, package_metadata_dir]
571            )
572            script_pre = f'{package_metadata_dir}/preinst'
573            script_post = f'{package_metadata_dir}/postinst'
574            # 1. preinst
575            if os.path.exists(script_pre):
576                Command.run(['chmod', '755', script_pre])
577                Command.run(
578                    [
579                        'chroot', self.root_dir,
580                        f'{script_pre.replace(self.root_dir, "")}', 'install'
581                    ], self.command_env
582                )
583            # 2. postinst
584            if os.path.exists(script_post):
585                Command.run(['chmod', '755', script_post])
586                Command.run(
587                    [
588                        'chroot', self.root_dir,
589                        f'{script_post.replace(self.root_dir, "")}', 'configure'
590                    ], self.command_env
591                )

181        ] + self.apt_get_args + self.custom_args + [
182            'update'
183        ]
184        install_command = [
185            'apt-get'
186        ] + self.apt_get_args + self.custom_args + [
187            'install', bootstrap_package
188        ]
189        Command.run(
190            update_command, self.command_env
191        )
192        Command.run(
193            install_command, self.command_env
194        )
195        # Unpack prebuilt bootstrap root tarball as new root
196        bootstrap_root_tarball = '/var/lib/bootstrap/{0}.{1}.tar.xz'.format(
197            bootstrap_package, defaults.PLATFORM_MACHINE
198        )
199        if not os.path.isfile(bootstrap_root_tarball):
200            raise KiwiFileNotFound(
201                f'bootstrap tarball: {bootstrap_root_tarball!r} not found'
202            )
203        Command.run(
204            ['tar', '-C', self.root_dir, '-xf', bootstrap_root_tarball]
205        )
206        # Install eventual bootstrap packages as standard system install
207        return self.process_install_requests()

209    def _process_install_requests_bootstrap(self) -> CommandCallT:
210        """
211        Process package install requests for bootstrap phase (no chroot).
212
213        :raises KiwiDebianBootstrapError
214
215        :return: process results in command type
216
217        :rtype: CommandCallT
218        """
219        # we invoke apt-get install to download all the essential packages.
220        # With DPkg::Pre-Install-Pkgs, we specify a shell command that will
221        # receive the list of packages that will be installed on stdin.
222        # By configuring Debug::pkgDpkgPm=1, apt-get install will not
223        # actually execute any dpkg commands, so all it does is download
224        # the essential debs and tell us their full in the apt cache without
225        # actually installing them.
226        try:
227            # TODO: Drop once apt 2.5.4 is widely available.
228            pathlib.Path(f'{self.root_dir}/var/lib/dpkg').mkdir(
229                parents=True, exist_ok=True
230            )
231            # Older versions of apt look for the status file both within and
232            # without the tree, but do not create the status file within the
233            # tree.
234            pathlib.Path(f'{self.root_dir}/var/lib/dpkg/status.kiwi').touch()
235            # Some versions of apt don't create the status file outside the
236            # tree, e.g. 2.4.14.
237            pathlib.Path('/var/lib/dpkg/status.kiwi').touch()
238            pathlib.Path(f'{self.root_dir}/var/lib/dpkg/available').touch()
239
240            if 'apt' not in self.package_requests:
241                self.package_requests.append('apt')
242            update_cache = [
243                'apt-get'
244            ] + self.apt_get_args + self.custom_args + [
245                'update'
246            ]
247            result = Command.run(
248                update_cache, self.command_env
249            )
250            log.debug(
251                'Apt update: {0} {1}'.format(result.output, result.error)
252            )
253            package_names = Temporary(prefix='kiwi_debs_').new_file()
254            self.bootstrap_packages = [
255                '?essential'
256            ] + self.package_requests
257            download_bootstrap = [
258                'apt-get'
259            ] + self.apt_get_args + self.custom_args + [
260                'install',
261                '-oDebug::pkgDPkgPm=1',
262                f'-oDPkg::Pre-Install-Pkgs::=cat >{package_names.name}'
263            ] + self.bootstrap_packages
264            # Download solved bootstrap packages
265            result = Command.run(
266                download_bootstrap, self.command_env
267            )
268            log.debug(
269                'Apt download: {0} {1}'.format(result.output, result.error)
270            )
271            # Extract bootstrap packages
272            with open(package_names.name) as packages:
273                solved_debootstrap_packages = [p.rstrip() for p in packages]
274            self.command_env['PATH'] = '$PATH:/usr/bin:/bin:/usr/sbin:/sbin'
275            for package in solved_debootstrap_packages:
276                Command.run(
277                    [
278                        'bash', '-c',
279                        'dpkg-deb --fsys-tarfile {0} | tar -C {1} -x'.format(
280                            package, self.root_dir
281                        )
282                    ], self.command_env
283                )
284            # Run package scripts. Unfortuantely Debian based systems
285            # requires special sauce for bootstrap. See the exceptions
286            # we have to apply below:
287            #
288            # * manual order is required to make sure users(root) exists
289            # * the usr-merge strategy only works after bootstrap
290            #
291            # 1. Pass: Run package scripts, manual order
292            self._run_bootstrap_scripts(
293                solved_debootstrap_packages,
294                only_for=['base-passwd'], skip=['usrmerge']
295            )
296            # 2. Pass: Run package scripts in apt order
297            self._run_bootstrap_scripts(
298                solved_debootstrap_packages,
299                skip=['usrmerge']
300            )
301            self.cleanup_requests()
302            return Command.call(
303                update_cache, self.command_env
304            )
305        except Exception as e:
306            raise KiwiDebianBootstrapError(
307                f'{type(e).__name__}: {format(e)}'
308            )

310    def process_install_requests(self) -> CommandCallT:
311        """
312        Process package install requests for image phase (chroot)
313
314        :return: process results in command type
315
316        :rtype: namedtuple
317        """
318        update_command = ['chroot', self.root_dir, 'apt-get']
319        update_command.extend(
320            Path.move_to_root(self.root_dir, self.apt_get_args)

324        Command.run(update_command, self.command_env)

333        # during bootstrap are not properly listed in the apt index.
334        # Therefore the bootstrap packages are added to the install

345    def process_delete_requests(self, force: bool = False) -> CommandCallT:
346        """
347        Process package delete requests (chroot)
348
349        :param bool force: force deletion: true|false
350
351        :raises KiwiRequestError: if none of the packages to delete is
352            installed
353        :return: process results in command type
354
355        :rtype: namedtuple
356        """
357        delete_items = []
358        for delete_item in self._package_requests():
359            try:
360                Command.run(
361                    ['chroot', self.root_dir, 'dpkg', '-l', delete_item]
362                )
363                delete_items.append(delete_item)
364            except Exception:
365                # ignore packages which are not installed
366                pass
367        if not delete_items:
368            raise KiwiRequestError(
369                'None of the requested packages to delete are installed'
370            )
371
372        # deleting debs for some reason can also trigger package
373        # scripts from the libc-bin (glibc) package. This often
374        # results in calls like ldconfig which failed unless the
375        # system would be really running. Since the deletion
376        # happens via chroot and with the system in offline mode
377        # many attempts to uninstall a package, even cracefully
378        # failed for this reason. So far I only saw the workaround
379        # to make ldconfig a noop during the process of uninstall.
380        # This is not a nice solution and I dislike it. However,
381        # the only one I could come up with to turn the package
382        # uninstall in a useful feature in kiwi.
383        Command.run(
384            [
385                'cp',
386                f'{self.root_dir}/usr/sbin/ldconfig',
387                f'{self.root_dir}/usr/sbin/ldconfig.orig'
388            ]
389        )
390        Command.run(
391            [
392                'cp',
393                f'{self.root_dir}/usr/bin/true',
394                f'{self.root_dir}/usr/sbin/ldconfig'
395            ]
396        )
397        if force:
398            # force deleting debs only worked well for me when ignoring
399            # the pre/post-inst and pre/post-remove codings. No idea why it
400            # ends in so many conflicts but if you want to force get rid
401            # of stuff this was the only way I could come up with. There
402            # are still cases when it does not work depending on the many
403            # code that runs on deleting
404            for package in delete_items:
405                pre_delete_pattern = \
406                    f'{self.root_dir}/var/lib/dpkg/info/{package}*.pre*'
407                post_delete_pattern = \
408                    f'{self.root_dir}/var/lib/dpkg/info/{package}*.post*'
409                for delete_file in glob.iglob(pre_delete_pattern):
410                    Path.wipe(delete_file)
411                for delete_file in glob.iglob(post_delete_pattern):
412                    Path.wipe(delete_file)
413
414            apt_get_command = ['chroot', self.root_dir, 'dpkg']
415            apt_get_command.extend(
416                [
417                    '--remove',
418                    '--force-remove-reinstreq',
419                    '--force-remove-essential',
420                    '--force-depends'
421                ]
422            )
423            apt_get_command.extend(delete_items)
424        else:
425            apt_get_command = ['chroot', self.root_dir, 'apt-get']
426            apt_get_command.extend(
427                Path.move_to_root(self.root_dir, self.apt_get_args)
428            )
429            apt_get_command.extend(self.custom_args)
430            apt_get_command.extend(['--auto-remove', 'remove'])
431            apt_get_command.extend(delete_items)
432
433        return Command.call(
434            apt_get_command, self.command_env
435        )

538    def _run_bootstrap_scripts(
539        self, solved_debootstrap_packages: List[str],
540        only_for: List[str] = [], skip: List[str] = []
541    ):
542        # TODO: this should not be needed but without setting
543        # the following environment variables no package pre/post
544        # script completes its task. I leave it up to the Debian
545        # experts to provide a fix if needed.
546        self.command_env['DPKG_MAINTSCRIPT_NAME'] = 'true'
547        self.command_env['DPKG_MAINTSCRIPT_PACKAGE'] = 'libc6'
548
549        post_script_dir = Temporary(
550            prefix='kiwi_debpost.', path=self.root_dir
551        ).new_dir()
552        for package in solved_debootstrap_packages:
553            package_base_name = os.path.basename(package)
554            go_ahead = False if only_for else True
555            for name in only_for:
556                if name in package_base_name:
557                    go_ahead = True
558                    break
559            for name in skip:
560                if name in package_base_name:
561                    go_ahead = False
562                    break
563            if not go_ahead:
564                continue
565            log.debug(
566                f'Running pre/post scripts for: {package_base_name}'
567            )
568            package_metadata_dir = \
569                f'{post_script_dir.name}/{os.path.basename(package)}'
570            Command.run(
571                ['dpkg', '-e', package, package_metadata_dir]
572            )
573            script_pre = f'{package_metadata_dir}/preinst'
574            script_post = f'{package_metadata_dir}/postinst'
575            # 1. preinst
576            if os.path.exists(script_pre):
577                Command.run(['chmod', '755', script_pre])
578                Command.run(
579                    [
580                        'chroot', self.root_dir,
581                        f'{script_pre.replace(self.root_dir, "")}', 'install'
582                    ], self.command_env
583                )
584            # 2. postinst
585            if os.path.exists(script_post):
586                Command.run(['chmod', '755', script_post])
587                Command.run(
588                    [
589                        'chroot', self.root_dir,
590                        f'{script_post.replace(self.root_dir, "")}', 'configure'
591                    ], self.command_env
592                )
