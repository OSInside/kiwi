 57class BootImageBase:
 58    """
 59    **Base class for boot image(initrd) task**
 60    """
 61    def __init__(
 62        self, xml_state: XMLState, target_dir: str,
 63        root_dir: Optional[str] = None,
 64        signing_keys: Optional[List[str]] = None,
 65        target_arch: Optional[str] = None
 66    ) -> None:
 67        """
 68        Create a base class instance of a BootImage
 69
 70        The base class instance only support the common operations
 71        for boot images. The actual preparation and creation of an
 72        initrd is specific to the used initrd creation toolchain
 73
 74        :param XMLState xml_state: Instance of :class:`XMLState`
 75        :param str target_dir: target dir to store the initrd
 76        :param str root_dir: system image root directory
 77        :param List[str] signing_keys: list of package signing keys
 78        :param str target_arch: target architecture
 79        """
 80        self.xml_state = xml_state
 81        self.target_dir = target_dir
 82        self.initrd_filename = ''
 83        self.boot_xml_state: Optional[XMLState] = None
 84        self.setup: Optional[SystemSetup] = None
 85        self.signing_keys = signing_keys
 86        self.target_arch = target_arch
 87        self.boot_root_directory = root_dir or ''
 88
 89        if not os.path.exists(target_dir):
 90            raise KiwiTargetDirectoryNotFound(
 91                f'target directory {target_dir} not found'
 92            )
 93
 94        self.initrd_base_name = ''.join(
 95            [
 96                self.xml_state.xml_data.get_name(),
 97                '.' + Defaults.get_platform_name(),
 98                '-' + self.xml_state.get_image_version(),
 99                '.initrd'
100            ]
101        )
102        self.post_init()

225    def get_boot_names(self) -> boot_names_type:
226        """
227        Provides kernel and initrd names for the boot image
228
229        :return:
230            Contains boot_names_type tuple
231
232            .. code:: python
233
234                boot_names_type(
235                    kernel_name='INSTALLED_KERNEL',
236                    initrd_name='DRACUT_OUTPUT_NAME',
237                    kernel_version='KERNEL_VERSION',
238                    kernel_filename='KERNEL_FILE_NAME'
239                )
240
241        :rtype: boot_names_type
242        """
243        kernel = Kernel(
244            self.boot_root_directory
245        )
246        kernel_info = kernel.get_kernel()
247        if not kernel_info:
248            if self.xml_state.get_initrd_system() == 'none':
249                return boot_names_type(
250                    kernel_name='none', initrd_name='none',
251                    kernel_version='none', kernel_filename='none'
252                )
253            raise KiwiDiskBootImageError(
254                f'No kernel in boot image tree {self.boot_root_directory} found'
255            )
256        dracut_output_format = self._get_boot_image_output_file_format(
257            kernel_info.version
258        )
259        return boot_names_type(
260            kernel_version=kernel_info.version,
261            kernel_name=kernel_info.name,
262            initrd_name=dracut_output_format.format(
263                kernel_version=kernel_info.version
264            ),
265            kernel_filename=kernel_info.filename
266        )

310    def load_boot_xml_description(self) -> None:
311        """
312        Load the boot image description referenced by the system image
313        description boot attribute
314        """
315        log.info('Loading Boot XML description')
316        boot_description_directory = self.get_boot_description_directory()
317        if not boot_description_directory:
318            raise KiwiConfigFileNotFound(
319                'no boot reference specified in XML description'
320            )
321        boot_config_file = boot_description_directory + '/config.xml'
322        if not os.path.exists(boot_config_file):
323            raise KiwiConfigFileNotFound(
324                f'no Boot XML description found in {boot_description_directory}'
325            )
326        boot_description = XMLDescription(
327            description=boot_config_file,
328            derived_from=self.xml_state.xml_data.description_dir
329        )
330
331        boot_image_profile = self.xml_state.build_type.get_bootprofile()
332        if not boot_image_profile:
333            boot_image_profile = 'default'
334        boot_kernel_profile = self.xml_state.build_type.get_bootkernel()
335        if not boot_kernel_profile:
336            boot_kernel_profile = 'std'
337
338        self.boot_xml_state = XMLState(
339            boot_description.load(), [boot_image_profile, boot_kernel_profile]
340        )
341        log.info('--> loaded %s', boot_config_file)
342        if self.boot_xml_state.build_type:
343            log.info(
344                '--> Selected build type: %s',
345                self.boot_xml_state.get_build_type_name()
346            )
347        if self.boot_xml_state.profiles:
348            log.info(
349                '--> Selected boot profiles: image: %s, kernel: %s',
350                boot_image_profile, boot_kernel_profile
351            )

460    def _get_boot_image_output_file_format(self, kernel_version: str) -> str:
461        """
462        The initrd output file format varies between
463        the different Linux distributions. Tools like lsinitrd, and also
464        grub2 rely on the initrd output file to be in that format. Thus
465        kiwi should use the same file format to stay compatible
466        with the distributions. The format is determined in three
467        stages:
468
469        a) check for an existing initrd file and use this naming schema
470        b) if no initrd file is found check the dracut binary for its
471           default output file name schema
472        c) if no output file schema could be detected from the dracut
473           binary return a default output file format which is
474           initramfs-{kernel_version}.img
475
476        Let's hope all this mess can be deleted once all distros just
477        can agree on one initrd file name schema
478        """
479        if self.xml_state.get_initrd_system() == 'kiwi':
480            # The custom kiwi initrd system is used only on SUSE systems.
481            # The initrd environment does not provide dracut and thus the
482            # outfile format cannot be determined. on SUSE systems the
483            # initrd format is different than on upstream and therefore
484            # it can be explicitly specified. Note that the custom initrd
485            # system will become obsolete in the future.
486            default_outfile_format = 'initrd-{kernel_version}'
487        else:
488            default_outfile_format = 'initramfs-{kernel_version}.img'
489
490        outfile_format = \
491            self._get_boot_image_output_file_format_from_existing_file(
492                kernel_version
493            )
494        if not outfile_format:
495            outfile_format = \
496                self._get_boot_image_output_file_format_from_dracut_code()
497
498        if not outfile_format:
499            outfile_format = \
500                self._get_boot_image_output_file_format_from_config(
501                    kernel_version
502                )
503
504        if outfile_format:
505            return outfile_format
506        else:
507            log.warning('Could not detect dracut output file format')
508            log.warning(
509                'Using default initrd file name format {0}'.format(
510                    default_outfile_format
511                )
512            )
513            return default_outfile_format
514
515    def _get_boot_image_output_file_format_from_dracut_code(
516        self
517    ) -> Optional[str]:
518        dracut_tool = Path.which(
519            'dracut', root_dir=self.boot_root_directory, access_mode=os.X_OK
520        )
521        if dracut_tool:
522            outfile_expression = r'outfile=".*/boot/(init.*kernel.*)"'
523            with open(dracut_tool) as dracut:
524                matches = re.findall(outfile_expression, dracut.read())
525                if matches:
526                    return matches[0].replace(
527                        '$kernel', '{kernel_version}'
528                    ).replace(
529                        '${kernel}', '{kernel_version}'
530                    )
531        return None
532
533    def _get_boot_image_output_file_format_from_existing_file(
534        self, kernel_version: str
535    ) -> Optional[str]:
536        for initrd_file in glob.iglob(self.boot_root_directory + '/boot/init*'):
537            if not os.path.islink(initrd_file):
538                return os.path.basename(initrd_file).replace(
539                    kernel_version, '{kernel_version}'
540                )
541        return None
542
543    def _get_boot_image_output_file_format_from_config(
544        self, kernel_version: str
545    ) -> Optional[str]:
546        if CommandCapabilities.has_option_in_help(
547            'dracut', '--printconfig', ['--help'],
548            root=self.boot_root_directory, raise_on_error=False
549        ):
550            dracut_call = Command.run(
551                ['chroot', self.boot_root_directory, 'dracut', '--printconfig']
552            )
553            temp_file = Temporary().new_file()
554            with open(temp_file.name, 'w') as config:
555                config.write(dracut_call.output.replace(' ', ''))
556            dracut_config = SysConfig(temp_file.name)
557            initrdname = dracut_config.get('dracutconfig:initrdname')
558            if initrdname:
559                return initrdname.replace(
560                    kernel_version, '{kernel_version}'
561                )
562        return None

461    def _get_boot_image_output_file_format(self, kernel_version: str) -> str:
462        """
463        The initrd output file format varies between
464        the different Linux distributions. Tools like lsinitrd, and also
465        grub2 rely on the initrd output file to be in that format. Thus
466        kiwi should use the same file format to stay compatible
467        with the distributions. The format is determined in three
468        stages:
469
470        a) check for an existing initrd file and use this naming schema
471        b) if no initrd file is found check the dracut binary for its
472           default output file name schema
473        c) if no output file schema could be detected from the dracut
474           binary return a default output file format which is
475           initramfs-{kernel_version}.img
476
477        Let's hope all this mess can be deleted once all distros just
478        can agree on one initrd file name schema
479        """
480        if self.xml_state.get_initrd_system() == 'kiwi':
481            # The custom kiwi initrd system is used only on SUSE systems.
482            # The initrd environment does not provide dracut and thus the
483            # outfile format cannot be determined. on SUSE systems the
484            # initrd format is different than on upstream and therefore
485            # it can be explicitly specified. Note that the custom initrd
486            # system will become obsolete in the future.
487            default_outfile_format = 'initrd-{kernel_version}'
488        else:
489            default_outfile_format = 'initramfs-{kernel_version}.img'
490
491        outfile_format = \
492            self._get_boot_image_output_file_format_from_existing_file(
493                kernel_version
494            )
495        if not outfile_format:
496            outfile_format = \
497                self._get_boot_image_output_file_format_from_dracut_code()
498
499        if not outfile_format:
500            outfile_format = \
501                self._get_boot_image_output_file_format_from_config(
502                    kernel_version
503                )
504
505        if outfile_format:
506            return outfile_format
507        else:
508            log.warning('Could not detect dracut output file format')
509            log.warning(
510                'Using default initrd file name format {0}'.format(
511                    default_outfile_format
512                )
513            )
514            return default_outfile_format
515
516    def _get_boot_image_output_file_format_from_dracut_code(
517        self
518    ) -> Optional[str]:
519        dracut_tool = Path.which(
520            'dracut', root_dir=self.boot_root_directory, access_mode=os.X_OK
521        )
522        if dracut_tool:
523            outfile_expression = r'outfile=".*/boot/(init.*kernel.*)"'
524            with open(dracut_tool) as dracut:
525                matches = re.findall(outfile_expression, dracut.read())
526                if matches:
527                    return matches[0].replace(
528                        '$kernel', '{kernel_version}'
529                    ).replace(
530                        '${kernel}', '{kernel_version}'
531                    )
532        return None
533
534    def _get_boot_image_output_file_format_from_existing_file(
535        self, kernel_version: str
536    ) -> Optional[str]:
537        for initrd_file in glob.iglob(self.boot_root_directory + '/boot/init*'):
538            if not os.path.islink(initrd_file):
539                return os.path.basename(initrd_file).replace(
540                    kernel_version, '{kernel_version}'
541                )
542        return None
543
544    def _get_boot_image_output_file_format_from_config(
545        self, kernel_version: str
546    ) -> Optional[str]:
547        if CommandCapabilities.has_option_in_help(
548            'dracut', '--printconfig', ['--help'],
549            root=self.boot_root_directory, raise_on_error=False
550        ):
551            dracut_call = Command.run(
552                ['chroot', self.boot_root_directory, 'dracut', '--printconfig']
553            )
554            temp_file = Temporary().new_file()
555            with open(temp_file.name, 'w') as config:
556                config.write(dracut_call.output.replace(' ', ''))
557            dracut_config = SysConfig(temp_file.name)
558            initrdname = dracut_config.get('dracutconfig:initrdname')
559            if initrdname:
560                return initrdname.replace(
561                    kernel_version, '{kernel_version}'
562                )
563        return None
