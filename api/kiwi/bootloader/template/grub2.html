253    def get_iso_template(
254        self, failsafe: bool = True,
255        has_graphics: bool = True, has_serial: bool = False,
256        checkiso: bool = False, has_btrfs_relative_path: bool = False
257    ):
258        """
259        Bootloader configuration template for live ISO media
260
261        :param bool failsafe: with failsafe true|false
262        :param bool has_graphics: supports graphics terminal
263        :param bool has_serial: supports serial terminal
264
265        :return: instance of :class:`Template`
266
267        :rtype: Template
268        """
269        if has_btrfs_relative_path:
270            template_data = self.header_btrfs_relative_path
271            template_data += self.header
272        else:
273            template_data = self.header
274        template_data += self.timeout
275        template_data += self.timeout_style
276        if has_graphics:
277            template_data += self.header_gfxmode
278            template_data += self.header_theme_iso
279        if has_serial:
280            template_data += self.header_serial
281        template_data += self.header_terminal_setup
282        template_data += self.menu_entry
283        if failsafe:
284            template_data += self.menu_entry_failsafe
285        if checkiso:
286            template_data += self.menu_mediacheck_entry
287        template_data += self.menu_iso_harddisk_entry
288        template_data += self.menu_entry_boot_snapshots
289        if has_graphics:
290            template_data += self.menu_entry_console_switch
291        return Template(template_data)
