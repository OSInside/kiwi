 42class BootLoaderSpecBase(BootLoaderConfigBase):
 43    """
 44    **Base class for Boot Loader Specification**
 45
 46    This base class follows the bootloader specifications as
 47    described in:
 48
 49    * https://uapi-group.org/specifications/specs/boot_loader_specification
 50
 51    All supported bootloaders which follows this specification
 52    are derived from this base class
 53
 54    :param object xml_state: instance of :class:`XMLState`
 55    :param str root_dir: root directory path name
 56    :param str boot_dir: boot directory path name
 57    :param dict custom_args:
 58        custom bootloader configuration parameters
 59    """
 60    def post_init(self, custom_args: Dict = {}) -> None:
 61        """
 62        Post init for all bootloader spec loaders
 63
 64        Store custom arguments in an instance dict and initialize
 65        target identifiers
 66        """
 67        self.entries_dir = Defaults.get_bls_loader_entries_dir()
 68        self.target = target_type(
 69            disk='disk', install='install(iso)', live='live(iso)'
 70        )
 71        self.custom_args = custom_args
 72        self.timeout = self.get_boot_timeout_seconds()
 73        self.disk_type = self.xml_state.get_build_type_bootloader_targettype()
 74        self.disk_blocksize = self.xml_state.build_type.get_target_blocksize()
 75        self.firmware = FirmWare(self.xml_state)
 76        self.cmdline = ''
 77
 78    def write(self) -> None:
 79        """
 80        For bootloaders following the bootloader spec
 81        no collective write will happen
 82        """
 83        pass
 84
 85    def setup_sysconfig_bootloader(self) -> None:
 86        """
 87        For bootloaders following the bootloader spec
 88        no setting to sysconfig/bootloader will be made
 89        """
 90        pass
 91
 92    def write_meta_data(
 93        self, root_device=None, write_device=None, boot_options=''
 94    ):
 95        """
 96        For bootloaders following the bootloader spec take over
 97        the cmdline options and store them for later use
 98
 99        :param string root_device: unused
100        :param string write_device: unused
101        :param string boot_options: kernel options as string
102        """
103        self.cmdline = f'{self.cmdline} {boot_options}'.strip()
104
105    def setup_disk_image_config(
106        self, boot_uuid: str = '', root_uuid: str = '', hypervisor: str = '',
107        kernel: str = '', initrd: str = '', boot_options: Dict = {}
108    ) -> None:
109        """
110        Create boot config file to boot from disk
111
112        :param str boot_uuid: boot device UUID
113        :param str root_uuid: root device UUID
114        :param str hypervisor: unused
115        :param str kernel: kernel name
116        :param str initrd: initrd name
117        :param dict boot_options:
118            custom options dictionary required to setup the bootloader.
119
120        Targeted to bootloader spec interface
121        """
122        self.custom_args['boot_uuid'] = boot_uuid
123        self.custom_args['root_uuid'] = root_uuid
124        self.custom_args['kernel'] = kernel
125        self.custom_args['initrd'] = initrd
126        self.custom_args['boot_options'] = boot_options
127        plus_cmdline = ' '.join(
128            [
129                self.get_boot_cmdline(
130                    boot_options.get('root_device'),
131                    boot_options.get('write_device')
132                )
133            ]
134        )
135        self.cmdline = f'{self.cmdline} {plus_cmdline}'.strip()
136        self.setup_loader(self.target.disk)

238        Setup bootloader menu entry boot/loader/entries/X.conf

243            target identifier, one of disk, live(iso) or install(iso)

247        raise NotImplementedError

249    def get_entry_name(self) -> str:
250        """
251        Construct entry filename of the form
252        [OS_release_ID]-[kernel-version].conf or use the
253        naming policy of an existing entries file
254
255        :return: file basename
256
257        :rtype: str
258        """
259        os_release = OsRelease(self.root_dir)
260        try:
261            kernel_name = os.path.basename(
262                list(glob.iglob(f'{self.root_dir}/lib/modules/*'))[0]
263            )
264        except Exception as issue:
265            raise KiwiKernelLookupError(
266                'Kernel lookup in {0}/lib/modules failed with: {1}'.format(
267                    self.root_dir, issue
268                )
269            )
270        entry_filename = f'{os_release.get("ID")}-{kernel_name}.conf'
271        # Check if there is already an entry file name produced by the
272        # distribution logic/policy. If we can find an already present
273        # entry name, prefer this name over the kiwi naming policy
274        for dist_entry in glob.iglob(f'{self.root_dir}/{self.entries_dir}/*'):
275            entry_filename = os.path.basename(dist_entry)
276            break
277        return entry_filename
