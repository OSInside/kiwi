 45class FileSystemBase:
 46    """
 47    **Implements base class for filesystem interface**
 48
 49    :param object device_provider:
 50        Instance of a class based on DeviceProvider
 51        required for filesystems which needs a block device for
 52        creation. In most cases the DeviceProvider is a LoopDevice
 53    :param string root_dir: root directory path name
 54    :param dict custom_args: custom filesystem arguments
 55    """
 56    def __init__(
 57        self, device_provider: DeviceProvider,
 58        root_dir: str = '', custom_args: Dict = {}
 59    ):
 60        # filesystems created with a block device stores the mountpoint
 61        # here. The file name of the file containing the filesystem is
 62        # stored in the device_provider if the filesystem is represented
 63        # as a file there
 64        self.filesystem_mount: Optional[MountManager] = None
 65
 66        # the underlaying device provider. This is only required if the
 67        # filesystem required a block device to become created
 68        self.device_provider = device_provider
 69
 70        self.root_dir = root_dir
 71
 72        # filesystems created without a block device stores the result
 73        # filesystem file name here
 74        self.filename = ''
 75
 76        self.custom_args: Dict = {}
 77        self.post_init(custom_args)
 78        self.veritysetup: Optional[VeritySetup] = None
 79
 80    def __enter__(self):
 81        return self

 83    def post_init(self, custom_args: Dict):
 84        """
 85        Post initialization method
 86
 87        Store dictionary of custom arguments if not empty. This
 88        overrides the default custom argument hash
 89
 90        :param dict custom_args: custom arguments
 91
 92            .. code:: python
 93
 94                {
 95                    'create_options': ['option'],
 96                    'mount_options': ['option'],
 97                    'meta_data': {
 98                        'key': 'value'
 99                    }
100                }
101        """
102        if custom_args:
103            self.custom_args = copy.deepcopy(custom_args)
104
105        if not self.custom_args.get('create_options'):
106            self.custom_args['create_options'] = []
107
108        if not self.custom_args.get('meta_data'):
109            self.custom_args['meta_data'] = {}
110
111        if not self.custom_args.get('mount_options'):
112            self.custom_args['mount_options'] = []
113
114        if not self.custom_args.get('fs_attributes'):
115            self.custom_args['fs_attributes'] = []

179    def sync_data(self, exclude: List[str] = []) -> MountManager:
180        """
181        Copy data tree into filesystem
182
183        :param list exclude: list of exclude dirs/files
184        :return: The mount created for syncing data. It should be used to
185            un-mount the filesystem again.
186        """
187        if not self.root_dir:
188            raise KiwiFileSystemSyncError(
189                'no root directory specified'
190            )
191        if not os.path.exists(self.root_dir):
192            raise KiwiFileSystemSyncError(
193                f'given root directory {self.root_dir} does not exist'
194            )
195        self.filesystem_mount = MountManager(
196            device=self.device_provider.get_device()
197        )
198        self.filesystem_mount.mount(
199            self.custom_args['mount_options']
200        )
201        self._apply_attributes()
202        data = DataSync(
203            self.root_dir, self.filesystem_mount.mountpoint
204        )
205        data.sync_data(
206            exclude=exclude, options=Defaults.get_sync_options()
207        )
208        return self.filesystem_mount

322    def _generate_seed_uuid(self, label: str, random_bits: int = 128) -> str:
323        """
324        Create random UUID. If SOURCE_DATE_EPOCH is present use
325        SOURCE_DATE_EPOCH + label name as seed
326        """
327        sde = os.environ.get('SOURCE_DATE_EPOCH')
328        if sde:
329            label_seed = reduce(lambda x, y: x + y, map(ord, label))
330            epoch_seed = label_seed + int(sde)
331            log.info(
332                'Using UUID seed SOURCE_DATE_EPOCH:{0} + LABEL:{1}={2}'.format(
333                    sde, label, label_seed
334                )
335            )
336            rd = random.Random()
337            rd.seed(epoch_seed)
338            return format(
339                uuid.UUID(int=rd.getrandbits(random_bits))
340            )
341        else:
342            return format(uuid.uuid4())

344    def _map_size(self, size: float, from_unit: str, to_unit: str) -> float:
345        """
346        Return byte size value for given size and unit
347
348        :param float size:
349            requested filesystem size. The value is interpreted
350            by the given from_unit.
351
352        :param str from_unit: source unit
353        :param str to_unit: target unit
354
355        :return: size value in unit: to_unit
356
357        :rtype: float
358        """
359        unit_map = {
360            defaults.UNIT.byte: 1,
361            defaults.UNIT.kb: 1024,
362            defaults.UNIT.mb: 1048576,
363            defaults.UNIT.gb: 1073741824
364        }
365        byte_size = size * unit_map[from_unit]
366        return byte_size / unit_map[to_unit]
367
368    def _fs_size(
369        self, size: float, blocksize: int = 1, unit: str = defaults.UNIT.kb
370    ) -> str:
371        """
372        Calculate filesystem size parameter in number of blocksize
373        blocks. If the given size is <= 0 the calculation is done from
374        the actual size of the block device reduced by the given size
375
376        :param float size:
377            requested filesystem size. The value is interpreted
378            by the given unit.
379
380        :param int blocksize:
381            blocksize as requested from the filesystem creation tool
382            for specifying the filesystem size. The value is interpreted
383            by the given unit. By default set to: 1
384
385        :param str unit:
386            Unit to use for calculations and return value
387            Default unit is set to: defaults.UNIT.kb
388
389        :return: an int block count of the specified unit as str
390
391        :rtype: str
392        """
393        if size > 0:
394            result_size = size / blocksize
395        else:
396            device_name = self.device_provider.get_device()
397            device_byte_size = self.device_provider.get_byte_size(device_name)
398            requested_byte_size = self._map_size(size, unit, defaults.UNIT.byte)
399            result_size = self._map_size(
400                (device_byte_size + requested_byte_size) / blocksize,
401                from_unit=defaults.UNIT.byte, to_unit=unit
402            )
403        return format(int(result_size))
404
405    def _apply_attributes(self):
406        """
407        Apply filesystem attributes
408        """
409        attribute_map = {
410            'synchronous-updates': '+S',
411            'no-copy-on-write': '+C'
412        }
413        for attribute in self.custom_args['fs_attributes']:
414            if attribute_map.get(attribute):
415                log.info(
416                    '--> setting {0} for {1}'.format(
417                        attribute, self.filesystem_mount.mountpoint
418                    )
419                )
420                Command.run(
421                    [
422                        'chattr', attribute_map.get(attribute),
423                        self.filesystem_mount.mountpoint
424                    ]
425                )
426
427    def __exit__(self, exc_type, exc_value, traceback):
428        self.umount()
