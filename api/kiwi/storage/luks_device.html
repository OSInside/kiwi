 39class LuksDevice(DeviceProvider):
 40    """
 41    **Implements luks setup on a storage device**
 42
 43    :param object storage_provider: Instance of class based on DeviceProvider
 44    """
 45    def __init__(self, storage_provider: DeviceProvider) -> None:
 46        #: the underlaying device provider
 47        self.storage_provider = storage_provider
 48
 49        self.luks_device: Optional[str] = None
 50        self.luks_keyfile: str = ''
 51        self.passphrase: str = ''
 52        self.luks_name = 'luksRoot'
 53
 54        self.option_map = {
 55            'sle12': [
 56                '--cipher', 'aes-xts-plain64',
 57                '--key-size', '256',
 58                '--hash', 'sha1'
 59            ]
 60        }
 61
 62    def __enter__(self):
 63        return self
 64
 65    def get_device(self) -> Optional[MappedDevice]:
 66        """
 67        Instance of MappedDevice providing the luks device
 68
 69        :return: mapped luks device
 70
 71        :rtype: MappedDevice
 72        """
 73        if self.luks_device:
 74            return MappedDevice(
 75                device=self.luks_device, device_provider=self
 76            )
 77        return None

 79    def create_crypto_luks(
 80        self, passphrase: str, osname: str = None,
 81        options: list = None, keyfile: str = '', randomize: bool = True,
 82        root_dir: str = ''
 83    ) -> None:
 84        """
 85        Create luks device. Please note the passphrase is readable
 86        at creation time of this image. Make sure your host system
 87        is secure while this process runs
 88
 89        :param string passphrase: credentials
 90        :param string osname:
 91            distribution name to match distribution specific
 92            options for cryptsetup
 93        :param list options: further cryptsetup options
 94        :param string keyfile: file path name
 95            file path name which contains an alternative key
 96            to unlock the luks device
 97        :param string root_dir: root dir path
 98        """
 99        keyslot = '0'
100        self.passphrase = passphrase
101        if not options:
102            options = []
103        if osname:
104            if osname in self.option_map:
105                options += self.option_map[osname]
106            else:
107                raise KiwiLuksSetupError(
108                    f'no custom option configuration found for OS {osname}'
109                )
110        extra_options = []
111        storage_device = self.storage_provider.get_device()
112        log.info('Creating crypto LUKS on %s', storage_device)
113
114        if not passphrase:
115            log.warning('Using an empty passphrase for the key setup')
116
117        if keyfile:
118            self.luks_keyfile = keyfile
119            keyfile_path = os.path.normpath(
120                os.sep.join([root_dir, self.luks_keyfile])
121            )
122            random_passphrase = LuksDevice.create_random_keyfile(keyfile_path)
123
124        if randomize:
125            log.info('--> Randomizing...')
126            storage_size_mbytes = self.storage_provider.get_byte_size(
127                storage_device
128            ) / 1048576
129            Command.run(
130                [
131                    'dd', 'if=/dev/urandom', 'bs=1M',
132                    'count=%d' % storage_size_mbytes,
133                    f'of={storage_device}'
134                ]
135            )
136
137        log.info('--> Creating LUKS map')
138
139        if passphrase and passphrase == 'random':
140            # In random mode use the generated keyfile as the only
141            # key to decrypt. This is only secure if the generated
142            # initrd also gets protected, e.g through encryption
143            # like it is done with the secure linux execution on
144            # zSystems
145            self.passphrase = random_passphrase
146            passphrase_file = keyfile_path
147            # Do not add an additional keyfile
148            keyfile = ''
149        elif passphrase:
150            # Setup a passphrase file for which the system will
151            # ask for in an interactive dialog
152            passphrase_file_tmp = Temporary().new_file()
153            with open(passphrase_file_tmp.name, 'w') as credentials:
154                credentials.write(passphrase)
155            passphrase_file = passphrase_file_tmp.name
156        else:
157            # Setup an empty passphrase, insecure and only useful
158            # for initial deployment which then applies a process
159            # to secure the image e.g reencrypt
160            passphrase_file_zero = '/dev/zero'
161            extra_options = [
162                '--keyfile-size', '32'
163            ]
164            passphrase_file = passphrase_file_zero
165
166        Command.run(
167            [
168                'cryptsetup', '-q', '--key-file', passphrase_file
169            ] + options + extra_options + [
170                'luksFormat', storage_device
171            ]
172        )
173
174        if keyfile:
175            Command.run(
176                [
177                    'cryptsetup', '--key-file', passphrase_file
178                ] + extra_options + [
179                    'luksAddKey', storage_device, keyfile_path
180                ]
181            )
182
183        # Create backup header checksum as reencryption reference
184        master_checksum = f'{root_dir}/root/.luks.header'
185        Path.wipe(master_checksum)
186        Command.run(
187            [
188                'cryptsetup', '--key-file', passphrase_file
189            ] + extra_options + [
190                'luksHeaderBackup', storage_device,
191                '--header-backup-file', master_checksum
192            ]
193        )
194        checksum = Checksum(master_checksum).sha256()
195        with open(master_checksum, 'w') as shasum:
196            shasum.write(checksum)
197
198        # Create key slot number as reencryption reference
199        master_slot = f'{root_dir}/root/.luks.slot'
200        with open(master_slot, 'w') as slot:
201            slot.write(keyslot)
202
203        # Create slot passphrase as reencryption reference
204        master_slotpass = f'{root_dir}/root/.slotpass'
205        with open(master_slotpass, 'w') as slotpass:
206            slotpass.write(self.passphrase)
207
208        # open the pool
209        Command.run(
210            [
211                'cryptsetup', '--key-file', passphrase_file
212            ] + extra_options + [
213                'luksOpen', storage_device, self.luks_name
214            ]
215        )
216        self.luks_device = '/dev/mapper/' + self.luks_name
217
218    def create_crypttab(self, filename: str) -> None:
219        """
220        Create crypttab, setting the UUID of the storage device
221
222        :param string filename: file path name
223        """
224        storage_device = self.storage_provider.get_device()
225        with open(filename, 'w') as crypttab:
226            luks_uuid = self.storage_provider.get_uuid(storage_device)
227            if self.luks_keyfile:
228                crypttab.write(
229                    'luks UUID={0} /{1}{2}'.format(
230                        luks_uuid, self.luks_keyfile.lstrip(os.sep), os.linesep
231                    )
232                )
233            else:
234                crypttab.write(
235                    'luks UUID={0}{1}'.format(
236                        luks_uuid, os.linesep
237                    )
238                )
239
240    def is_loop(self) -> bool:
241        """
242        Check if storage provider is loop based
243
244        Return loop status from base storage provider
245
246        :return: True or False
247
248        :rtype: bool
249        """
250        return self.storage_provider.is_loop()
251
252    @staticmethod
253    def create_random_keyfile(filename: str) -> str:
254        """
255        Create keyfile with random data
256
257        :param string filename: file path name
258        """
259        random_data = binascii.hexlify(
260            os.urandom(Defaults.get_luks_key_length())
261        ).decode()
262        with open(filename, 'w') as keyfile:
263            keyfile.write(random_data)
264        os.chmod(filename, 0o600)
265        return random_data
266
267    def __exit__(self, exc_type, exc_value, traceback):
268        if self.luks_device:
269            try:
270                Command.run(
271                    ['cryptsetup', 'luksClose', self.luks_name]
272                )
273            except Exception as issue:
274                log.error(
275                    'Shutdown of luks map {0}:{1} failed with: {2}'.format(
276                        self.luks_name, self.luks_device, issue
277                    )
278                )
