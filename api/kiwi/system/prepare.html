 55class SystemPrepare:
 56    """
 57    Implements preparation and installation of a new root system
 58
 59    :param object xml_state: instance of :class:`XMLState`
 60    :param str root_dir: Path to new root directory
 61    :param bool allow_existing: Allow using an existing root_dir
 62    """
 63    def __init__(
 64        self, xml_state: XMLState, root_dir: str, allow_existing: bool = False
 65    ):
 66        """
 67        Setup and host bind new root system at given root_dir directory
 68        """
 69        self.root_import = None
 70
 71        log.info('Setup root directory: %s', root_dir)
 72        if not log.getLogLevel() == logging.DEBUG and not log.get_logfile():
 73            self.issue_message = dedent('''
 74                {headline}: {reason}
 75
 76                Further details to clarify the error might have been
 77                reported earlier in the package manager log information
 78                and did not get exposed to the caller. Thus if the above
 79                message is not clear on the error please call kiwi with
 80                the --debug or --logfile option.
 81            ''')
 82        else:
 83            self.issue_message = '{headline}: {reason}'
 84        root = RootInit(
 85            root_dir, allow_existing
 86        )
 87        root.create()
 88        image_uris = xml_state.get_derived_from_image_uri()
 89        delta_root = xml_state.build_type.get_delta_root()
 90
 91        if image_uris:
 92            self.root_import = RootImport.new(
 93                root_dir, image_uris, xml_state.build_type.get_image()
 94            )
 95            if delta_root:
 96                self.root_import.overlay_data()
 97            else:
 98                self.root_import.sync_data()
 99        root_bind = RootBind(
100            root
101        )
102        root_bind.setup_intermediate_config()
103        root_bind.mount_kernel_file_systems(delta_root)
104        root_bind.mount_shared_directory()
105
106        self.delta_root = delta_root
107        self.root_dir = root_dir
108        self.xml_state = xml_state
109        self.profiles = xml_state.profiles
110        self.root_bind = root_bind
111
112        #: A list of Uri references
113        self.uri_list: List[Uri] = []

118    def setup_repositories(
119        self, clear_cache: bool = False,
120        signing_keys: List[str] = None, target_arch: Optional[str] = None
121    ) -> PackageManagerBase:
122        """
123        Set up repositories for software installation and return a
124        package manager for performing software installation tasks
125
126        :param bool clear_cache:
127            Flag the clear cache before configure anything
128        :param list signing_keys:
129            Keys imported to the package manager
130        :param str target_arch:
131            Target architecture name
132
133        :return: instance of :class:`PackageManager` subclass
134
135        :rtype: PackageManager
136        """
137        repository_options = []
138        repository_sections = \
139            self.xml_state.get_repository_sections_used_for_build()
140        package_manager = self.xml_state.get_package_manager()
141        release_version = self.xml_state.get_release_version()
142        rpm_locale_list = self.xml_state.get_rpm_locale()
143        if self.xml_state.get_rpm_check_signatures():
144            repository_options.append('check_signatures')
145        if self.xml_state.get_rpm_excludedocs():
146            repository_options.append('exclude_docs')
147        if rpm_locale_list:
148            repository_options.append(
149                '_install_langs%{0}'.format(':'.join(rpm_locale_list))
150            )
151        if target_arch:
152            repository_options.append(
153                f'_target_arch%{target_arch}'
154            )
155        with Repository.new(
156            self.root_bind, package_manager, repository_options
157        ) as repo:
158            repo.setup_package_database_configuration()
159            if signing_keys:
160                repo.import_trusted_keys(signing_keys)
161            for xml_repo in repository_sections:
162                repo_type = xml_repo.get_type()
163                repo_source = xml_repo.get_source().get_path()
164                repo_architectures = xml_repo.get_architectures()
165                repo_user = xml_repo.get_username()
166                repo_secret = xml_repo.get_password()
167                repo_alias = xml_repo.get_alias()
168                repo_priority = xml_repo.get_priority()
169                repo_dist = xml_repo.get_distribution()
170                repo_components = xml_repo.get_components()
171                repo_repository_gpgcheck = xml_repo.get_repository_gpgcheck()
172                repo_package_gpgcheck = xml_repo.get_package_gpgcheck()
173                repo_customization_script = self._get_repo_customization_script(
174                    xml_repo
175                )
176                repo_sourcetype = xml_repo.get_sourcetype()
177                log.info(
178                    'Setting up repository %s', Uri.print_sensitive(repo_source)
179                )
180                log.info('--> Type: {0}'.format(repo_type))
181                if repo_sourcetype:
182                    log.info('--> SourceType: {0}'.format(repo_sourcetype))
183                if repo_priority:
184                    log.info('--> Priority: {0}'.format(repo_priority))
185
186                uri = Uri(repo_source, repo_type)
187                repo_source_translated = uri.translate()
188                log.info(
189                    '--> Translated: {0}'.format(
190                        Uri.print_sensitive(repo_source_translated)
191                    )
192                )
193                if not repo_alias:
194                    repo_alias = uri.alias()
195                log.info('--> Alias: {0}'.format(repo_alias))
196
197                if not uri.is_remote() and not os.path.exists(
198                    repo_source_translated
199                ):
200                    log.warning(
201                        'repository %s does not exist and will be skipped',
202                        repo_source
203                    )
204                    continue
205
206                if not uri.is_remote():
207                    self.root_bind.mount_shared_directory(
208                        repo_source_translated
209                    )
210
211                repo.add_repo(
212                    repo_alias, repo_source_translated,
213                    repo_type, repo_priority, repo_dist, repo_components,
214                    repo_user, repo_secret, uri.credentials_file_name(),
215                    repo_repository_gpgcheck, repo_package_gpgcheck,
216                    repo_sourcetype, repo_customization_script,
217                    repo_architectures
218                )
219                if clear_cache:
220                    repo.delete_repo_cache(repo_alias)
221                self.uri_list.append(uri)
222            repo.cleanup_unused_repos()
223            return PackageManager.new(
224                repository=repo,
225                package_manager_name=package_manager,
226                release_version=release_version
227            )

229    def install_bootstrap(
230        self, manager: PackageManagerBase, plus_packages: List = None
231    ) -> None:
232        """
233        Install system software using the package manager
234        from the host, also known as bootstrapping
235
236        :param object manager: instance of a :class:`PackageManager` subclass
237        :param list plus_packages: list of additional packages
238
239        :raises KiwiBootStrapPhaseFailed:
240            if the bootstrapping process fails either installing
241            packages or including bootstrap archives
242        """
243        if not self.xml_state.get_bootstrap_packages_sections() \
244           and not plus_packages:
245            log.warning('No <packages> sections marked as "bootstrap" found')
246            log.info('Processing of bootstrap stage skipped')
247            return
248
249        log.info('Installing bootstrap packages')
250        bootstrap_packages = self.xml_state.get_bootstrap_packages(
251            plus_packages
252        )
253        collection_type = self.xml_state.get_bootstrap_collection_type()
254        log.info('--> collection type: %s', collection_type)
255        bootstrap_collections = self.xml_state.get_bootstrap_collections()
256        bootstrap_products = self.xml_state.get_bootstrap_products()
257        bootstrap_archives = self.xml_state.get_bootstrap_archives()
258        bootstrap_archives_target_dirs = self.xml_state.get_bootstrap_archives_target_dirs()
259        bootstrap_packages_ignored = self.xml_state.get_bootstrap_ignore_packages()
260        package_manager = self.xml_state.get_package_manager()
261        # process package installations
262        if collection_type == 'onlyRequired':
263            manager.process_only_required()
264        else:
265            manager.process_plus_recommended()
266        all_install_items = self._setup_requests(
267            manager,
268            bootstrap_packages,
269            bootstrap_collections,
270            bootstrap_products,
271            bootstrap_packages_ignored
272        )
273        manager.setup_repository_modules(
274            self.xml_state.get_collection_modules()
275        )
276        process = CommandProcess(
277            command=manager.process_install_requests_bootstrap(
278                self.root_bind, self.xml_state.get_bootstrap_package_name()
279            ), log_topic='bootstrap'
280        )
281        try:
282            process.poll_show_progress(
283                items_to_complete=all_install_items,
284                match_method=process.create_match_method(
285                    manager.match_package_installed
286                ),
287                with_stderr=True if package_manager == 'dnf5' else False
288            )
289        except Exception as issue:
290            if manager.has_failed(process.returncode()):
291                raise KiwiBootStrapPhaseFailed(
292                    self.issue_message.format(
293                        headline='Bootstrap package installation failed',
294                        reason=f'{issue}: {manager.get_error_details()}'
295                    )
296                )
297        manager.post_process_install_requests_bootstrap(
298            self.root_bind, self.delta_root
299        )
300        # process archive installations
301        if bootstrap_archives:
302            try:
303                self._install_archives(
304                    bootstrap_archives, bootstrap_archives_target_dirs
305                )
306            except Exception as issue:
307                raise KiwiBootStrapPhaseFailed(
308                    self.issue_message.format(
309                        headline='Bootstrap archive installation failed',
310                        reason=issue
311                    )
312                )

314    def install_system(self, manager: PackageManagerBase) -> None:
315        """
316        Install system software using the package manager inside
317        of the new root directory. This is done via a chroot operation
318        and requires the desired package manager to became installed
319        via the bootstrap phase
320
321        :param object manager: instance of a :class:`PackageManager` subclass
322
323        :raises KiwiInstallPhaseFailed:
324            if the install process fails either installing
325            packages or including any archive
326        """
327        log.info(
328            'Installing system (chroot) for build type: %s',
329            self.xml_state.get_build_type_name()
330        )
331        collection_type = self.xml_state.get_system_collection_type()
332        log.info('--> collection type: %s', collection_type)
333        system_packages = self.xml_state.get_system_packages()
334        system_collections = self.xml_state.get_system_collections()
335        system_products = self.xml_state.get_system_products()
336        system_archives = self.xml_state.get_system_archives()
337        system_archives_target_dirs = self.xml_state.get_system_archives_target_dirs()
338        system_packages_ignored = self.xml_state.get_system_ignore_packages()
339        package_manager = self.xml_state.get_package_manager()
340        # process package installations
341        if collection_type == 'onlyRequired':
342            manager.process_only_required()
343        else:
344            manager.process_plus_recommended()
345        all_install_items = self._setup_requests(
346            manager,
347            system_packages,
348            system_collections,
349            system_products,
350            system_packages_ignored
351        )
352        if all_install_items:
353            process = CommandProcess(
354                command=manager.process_install_requests(), log_topic='system'
355            )
356            try:
357                process.poll_show_progress(
358                    items_to_complete=all_install_items,
359                    match_method=process.create_match_method(
360                        manager.match_package_installed
361                    ),
362                    with_stderr=True if package_manager == 'dnf5' else False
363                )
364            except Exception as issue:
365                if manager.has_failed(process.returncode()):
366                    raise KiwiInstallPhaseFailed(
367                        self.issue_message.format(
368                            headline='System package installation failed',
369                            reason=issue
370                        )
371                    )
372        # process archive installations
373        if system_archives:
374            try:
375                self._install_archives(
376                    system_archives,
377                    system_archives_target_dirs
378                )
379            except Exception as issue:
380                raise KiwiInstallPhaseFailed(
381                    self.issue_message.format(
382                        headline='System archive installation failed',
383                        reason=issue
384                    )
385                )

387    def pinch_system(
388        self, manager: Optional[PackageManagerBase] = None, force: bool = False
389    ) -> None:
390        """
391        Delete packages marked for deletion in the XML description. If force
392        param is set to False uninstalls packages marked with
393        `type="uninstall"` if any; if force is set to True deletes packages
394        marked with `type="delete"` if any.
395
396        :param object manager: instance of :class:`PackageManager` subclass
397        :param bool force: Forced deletion True|False
398
399        :raises KiwiPackagesDeletePhaseFailed:
400            if the deletion packages process fails
401        """
402        if self.delta_root:
403            # In delta mode create a reference tree to allow
404            # to diff on deleted data
405            Command.run(
406                [
407                    'rsync', '-a',
408                    f'{self.root_dir}_cow/',
409                    f'{self.root_dir}_cow_before_pinch'
410                ]
411            )
412        to_become_deleted_packages = \
413            self.xml_state.get_to_become_deleted_packages(force)
414        if to_become_deleted_packages:
415            log.info(
416                '{0} system packages (chroot)'.format(
417                    'Force deleting' if force else 'Uninstalling'
418                )
419            )
420            try:
421                with ExitStack() as stack:
422                    if manager is None:
423                        package_manager = self.xml_state.get_package_manager()
424                        release_version = self.xml_state.get_release_version()
425                        repo = Repository.new(self.root_bind, package_manager)
426                        stack.push(repo)
427                        manager = PackageManager.new(
428                            repository=repo,
429                            package_manager_name=package_manager,
430                            release_version=release_version
431                        )
432                        stack.push(manager)
433                    self.delete_packages(
434                        manager, to_become_deleted_packages, force
435                    )
436            except Exception as issue:
437                raise KiwiPackagesDeletePhaseFailed(
438                    self.issue_message.format(
439                        headline='System package deletion failed',
440                        reason=issue
441                    )
442                )

481    def delete_packages(
482        self, manager: PackageManagerBase, packages: List, force: bool = False
483    ) -> None:
484        """
485        Delete one or more packages using the package manager inside
486        of the new root directory. If the removal is set with `force` flag
487        only listed packages are deleted and any dependency break or leftover
488        is ignored.
489
490        :param object manager: instance of a :class:`PackageManager` subclass
491        :param list packages: package list
492        :param bool force: force deletion true|false
493
494        :raises KiwiSystemDeletePackagesFailed: if installation process fails
495        """
496        package_manager = self.xml_state.get_package_manager()
497        all_delete_items = self._setup_requests(
498            manager, packages
499        )
500        if all_delete_items:
501            log.info(
502                '{0} system packages (chroot)'.format(
503                    'Force deleting' if force else 'Uninstall'
504                )
505            )
506            process = CommandProcess(
507                command=manager.process_delete_requests(force),
508                log_topic='system'
509            )
510            try:
511                process.poll_show_progress(
512                    items_to_complete=all_delete_items,
513                    match_method=process.create_match_method(
514                        manager.match_package_deleted
515                    ),
516                    with_stderr=True if package_manager == 'dnf5' else False
517                )
518                manager.post_process_delete_requests(self.root_bind)
519            except Exception as issue:
520                raise KiwiSystemDeletePackagesFailed(
521                    self.issue_message.format(
522                        headline='Package deletion failed',
523                        reason=issue
524                    )
525                )

566    def setup_ca_certificates(self) -> None:
567        """
568        Setup custom CA certificates in the chroot environment. This is
569        done by copying the certificate list from get_certificates()
570        to the distribution specific certificate store followed by
571        calling the distribution specific update tool.
572
573        :raises KiwiBootStrapPhaseFailed:
574            The CA certificates are added immediately after the bootstrap
575            package installation, yet before the system packages are
576            installed, therefore a bootstrap failure is raised if we
577            catch any exception here.
578        """
579        ca_certs = self.xml_state.get_certificates()
580        if ca_certs:
581            # Determine the correct distro specific update tool and cert store
582            log.info('Updating CA certificate store')
583            target_distribution = \
584                self.xml_state.get_certificates_target_distribution()
585            ca_update_info = Defaults.get_ca_update_map(target_distribution)
586            if ca_update_info:
587                ca_chroot_path = os.path.normpath(
588                    os.sep.join(
589                        [self.root_dir, ca_update_info['destination_path']]
590                    )
591                )
592                # sync certificates, we accept no errors on this operation
593                try:
594                    Path.create(ca_chroot_path)
595                    has_new_certs = False
596                    for cert_file in ca_certs:
597                        if os.path.exists(cert_file) \
598                           and not os.path.isdir(cert_file):
599                            log.info(
600                                f'--> Syncing: {cert_file} to {ca_chroot_path}'
601                            )
602                            DataSync(cert_file, ca_chroot_path).sync_data()
603                            has_new_certs = True
604                        else:
605                            log.warning(
606                                f'--> Skipping: {cert_file}: '
607                                'does not exist or is directory'
608                            )
609                    if has_new_certs:
610                        ca_tool = ca_update_info['tool']
611                        log.info(
612                            f'--> Updating CA certificate store with: {ca_tool}'
613                        )
614                        Command.run(
615                            ['chroot', self.root_dir, ca_tool]
616                        )
617                except Exception as issue:
618                    raise KiwiBootStrapPhaseFailed(
619                        self.issue_message.format(
620                            headline='Failed to setup custom CA certificates',
621                            reason=issue
622                        )
623                    )
624            else:
625                log.warning(
626                    'Could not determine CA update tool, skipping setup'
627                )

629    def _install_archives(self, archive_list, archive_target_dir_dict):
630        log.info("Installing archives")
631        for archive in archive_list:
632            log.info("--> archive: %s", archive)
633            description_dir = \
634                self.xml_state.xml_data.description_dir
635            derived_description_dir = \
636                self.xml_state.xml_data.derived_description_dir
637            archive_is_absolute = archive.startswith('/')
638            if archive_is_absolute:
639                archive_file = archive
640            else:
641                archive_file = '/'.join(
642                    [description_dir, archive]
643                )
644            archive_exists = os.path.exists(archive_file)
645            if not archive_is_absolute \
646               and not archive_exists and derived_description_dir:
647                archive_file = '/'.join(
648                    [derived_description_dir, archive]
649                )
650            target_dir = self.root_bind.root_dir
651            if archive_target_dir_dict.get(archive):
652                target_dir = os.path.normpath(
653                    os.sep.join(
654                        [target_dir, archive_target_dir_dict.get(archive)]
655                    )
656                )
657            log.info('--> target dir: %s', target_dir)
658            tar = ArchiveTar(archive_file)
659            tar.extract(target_dir)

661    def _setup_requests(
662        self, manager, packages, collections=None, products=None, ignored=None
663    ):
664        if packages:
665            for package in sorted(packages):
666                log.info('--> package: {0}'.format(package))
667                manager.request_package(package)
668        if collections:
669            for collection in sorted(collections):
670                log.info('--> collection: {0}'.format(collection))
671                manager.request_collection(collection)
672        if products:
673            for product in sorted(products):
674                log.info('--> product: {0}'.format(product))
675                manager.request_product(product)
676        if ignored:
677            for package in sorted(ignored):
678                log.info('--> package excluded: {0}'.format(package))
679                manager.request_package_exclusion(package)
680        return \
681            manager.package_requests + \
682            manager.collection_requests + \
683            manager.product_requests + \
684            manager.exclude_requests

686    def _get_repo_customization_script(self, xml_repo: repository) -> str:
687        script_path = xml_repo.get_customize()
