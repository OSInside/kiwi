 37class RepositoryApt(RepositoryBase):
 38    """
 39    **Implements repository handling for apt-get package manager**
 40
 41    :param str shared_apt_get_dir:
 42        shared directory between image root and build system root
 43    :param str runtime_apt_get_config_file: apt-get runtime config file name
 44    :param list apt_get_args: apt-get caller arguments
 45    :param dict command_env: customized os.environ for apt-get
 46    :param manager_base: shared location for apt-get repodata between
 47        host and image
 48    """
 49
 50    def post_init(self, custom_args: List = []) -> None:
 51        """
 52        Post initialization method
 53
 54        Store custom apt-get arguments and create runtime configuration
 55        and environment
 56
 57        :param list custom_args: apt-get arguments
 58        """
 59        self.runtime_apt_get_config_file = TmpT(name='')
 60        self.custom_args = custom_args
 61        self.exclude_docs = False
 62        self.signing_keys: List = []
 63
 64        # extract custom arguments used for apt config only
 65        if 'exclude_docs' in self.custom_args:
 66            self.custom_args.remove('exclude_docs')
 67            self.exclude_docs = True
 68
 69        if 'check_signatures' in self.custom_args:
 70            self.custom_args.remove('check_signatures')
 71            self.unauthenticated = 'false'
 72        else:
 73            self.unauthenticated = 'true'
 74
 75        self.distribution: str = ''
 76        self.distribution_path: str = ''
 77        self.repo_names: List = []
 78        self.components: List = []
 79
 80        # apt-get support is based on creating a sources file which
 81        # contains path names to the repo and its cache. In order to
 82        # allow a persistent use of the files in and outside of a chroot
 83        # call an active bind mount from RootBind::mount_shared_directory
 84        # is expected and required
 85        self.manager_base = self.shared_location + '/apt-get'
 86
 87        self.shared_apt_get_dir = {
 88            'sources-dir': self.manager_base + '/sources.list.d',
 89            'preferences-dir': self.manager_base + '/preferences.d'
 90        }
 91        self.keyring = '{}/trusted.gpg'.format(self.manager_base)
 92
 93        self.runtime_apt_get_config_file = Temporary(
 94            path=self.root_dir, prefix='kiwi_apt.config'
 95        ).unmanaged_file()
 96
 97        self.apt_get_args = [
 98            '-q', '-c', self.runtime_apt_get_config_file.name, '-y'
 99        ] + self.custom_args
100
101        self.command_env = self._create_apt_get_runtime_environment()
102
103        # config file for apt-get tool
104        self.apt_conf = PackageManagerTemplateAptGet()
105        self._write_runtime_config()
106
107    def setup_package_database_configuration(self) -> None:
108        """
109        Setup package database configuration
110
111        No special database configuration required for apt
112        """
113        pass
114
115    def use_default_location(self) -> None:
116        """
117        Setup apt-get repository operations to store all data
118        in the default places
119        """
120        self.manager_base = os.sep.join([self.root_dir, 'etc/apt'])
121        self.shared_apt_get_dir['sources-dir'] = \
122            os.sep.join([self.manager_base, 'sources.list.d'])
123        self.shared_apt_get_dir['preferences-dir'] = \
124            os.sep.join([self.manager_base, 'preferences.d'])
125        self._write_runtime_config(system_default=True)
126
127    def runtime_config(self) -> Dict:
128        """
129        apt-get runtime configuration and environment
130        """
131        return {
132            'apt_get_args': self.apt_get_args,
133            'command_env': self.command_env,
134            'distribution': self.distribution,
135            'distribution_path': self.distribution_path
136        }

138    def add_repo(
139        self, name: str, uri: str, repo_type: str = 'deb',
140        prio: int = None, dist: str = None, components: str = None,
141        user: str = None, secret: str = None, credentials_file: str = None,
142        repo_gpgcheck: bool = None, pkg_gpgcheck: bool = None,
143        sourcetype: str = None, customization_script: str = None,
144        architectures: str = None
145    ) -> None:
146        """
147        Add apt_get repository
148
149        :param str name: repository base file name
150        :param str uri: repository URI
151        :param str repo_type: unused
152        :param int prio: unused
153        :param str dist: distribution name for non flat deb repos
154        :param str components: distribution categories
155        :param str user: unused
156        :param str secret: unused
157        :param str credentials_file: unused
158        :param bool repo_gpgcheck: enable repository signature validation
159        :param bool pkg_gpgcheck: unused
160        :param str sourcetype: unused
161        :param str customization_script:
162            custom script called after the repo file was created
163        :param str architectures:
164            identifies which architectures are supported by this repository
165        """
166        sources_file = '/'.join(
167            [self.shared_apt_get_dir['sources-dir'], name + '.sources']
168        )
169        pref_file = '/'.join(
170            [self.shared_apt_get_dir['preferences-dir'], name + '.pref']
171        )
172        self.repo_names.append(name + '.sources')
173        if os.path.exists(uri):
174            # apt-get requires local paths to take the file: type
175            uri = 'file:/' + uri
176            uri = uri.replace('file://', 'file:/')
177        if not components:
178            components = 'main'
179        self._add_components(components)
180        with open(sources_file, 'w') as repo:
181            repo_details = 'Types: deb' + os.linesep
182            repo_details += 'URIs: ' + uri + os.linesep
183            if architectures:
184                repo_details += 'Architectures: {}{}'.format(
185                    architectures.replace(',', ' '), os.linesep
186                )
187            if not dist:
188                # create a debian flat repository setup. We consider the
189                # repository metadata to exist on the toplevel of the
190                # specified uri. This applies to the way the open build
191                # service creates debian repositories and should be
192                # done in the same way for other repositories when used
193                # with kiwi
194                repo_details += 'Suites: ./' + os.linesep
195            else:
196                # create a debian distributon repository setup for the
197                # specified distributon name and components
198                self.distribution = dist
199                self.distribution_path = uri
200                repo_details += 'Suites: ' + dist + os.linesep
201                repo_details += 'Components: ' + components + os.linesep
202            if repo_gpgcheck is False:
203                repo_details += 'trusted: yes' + os.linesep
204                repo_details += 'check-valid-until: no' + os.linesep
205            repo.write(repo_details)
206        if customization_script:
207            self.run_repo_customize(customization_script, sources_file)
208        if prio:
209            uri_parsed = urlparse(uri.replace('file://', 'file:/'))
210            with open(pref_file, 'w') as pref:
211                pref.write('Package: *{0}'.format(os.linesep))
212                if not uri_parsed.hostname:
213                    pref.write(
214                        'Pin: origin ""{0}'.format(os.linesep)
215                    )
216                else:
217                    pref.write(
218                        'Pin: origin "{0}"{1}'.format(
219                            uri_parsed.hostname, os.linesep
220                        )
221                    )
222                pref.write(
223                    'Pin-Priority: {0}{1}'.format(prio, os.linesep)
224                )
225            if customization_script:
226                self.run_repo_customize(customization_script, pref_file)
227
228    def import_trusted_keys(self, signing_keys: List) -> None:
229        """
230        Creates a new keyring including provided keys
231
232        :param list signing_keys: list of the key files to import
233        """
234        keybox = '{}/trusted-keybox.gpg'.format(self.manager_base)
235        gpg_args = [
236            'gpg', '--no-options', '--no-default-keyring',
237            '--no-auto-check-trustdb', '--trust-model', 'always',
238            '--keyring', keybox
239        ]
240        if os.path.exists(self.keyring):
241            os.unlink(self.keyring)
242        if os.path.exists(keybox):
243            os.unlink(keybox)
244        for key in signing_keys:
245            Command.run(gpg_args + ['--import', '--ignore-time-conflict', key])
246        if os.path.exists(keybox):
247            Command.run(
248                gpg_args + ['--export', '--yes', '--output', self.keyring]
249            )
250            os.unlink(keybox)
251            log.info('Custom keyring for APT created: {}'.format(self.keyring))
252            log.warning(
253                'The keyring is only available at build time. '
254                'It will not be part of the resulting image'
255            )

257    def delete_repo(self, name: str) -> None:
258        """
259        Delete apt-get repository
260
261        :param str name: repository base file name
262        """
263        Path.wipe(
264            self.shared_apt_get_dir['sources-dir'] + '/' + name + '.sources'
265        )

267    def delete_all_repos(self) -> None:
268        """
269        Delete all apt-get repositories
270        """
271        Path.wipe(self.shared_apt_get_dir['sources-dir'])
272        Path.create(self.shared_apt_get_dir['sources-dir'])
273
274    def delete_repo_cache(self, name: str) -> None:
275        """
276        Delete apt-get repository cache
277
278        Apt stores the package cache in a collection of binary files
279        and deb archives. As of now I couldn't came across a solution
280        which allows for deleting only the cache data for a specific
281        repository. Thus the repo cache cleanup affects all cache
282        data
283
284        :param str name: unused
285        """
286        for cache_file in ['archives', 'pkgcache.bin', 'srcpkgcache.bin']:
287            Path.wipe(os.sep.join([self.manager_base, cache_file]))
288
289    def cleanup_unused_repos(self) -> None:
290        """
291        Delete unused apt_get repositories
292
293        Repository configurations which are not used for this build
294        must be removed otherwise they are taken into account for
295        the package installations
296        """
297        repos_dir = self.shared_apt_get_dir['sources-dir']
298        repo_files = list(os.walk(repos_dir))[0][2]
299        for repo_file in repo_files:
300            if repo_file not in self.repo_names:
301                Path.wipe(repos_dir + '/' + repo_file)
302
303    def _add_components(self, components: str) -> None:
304        for component in components.split():
305            if component not in self.components:
306                self.components.append(component)
307
308    def _create_apt_get_runtime_environment(self) -> Dict:
309        for apt_get_dir in list(self.shared_apt_get_dir.values()):
310            Path.create(apt_get_dir)
311        ToEnv(self.root_dir, defaults.PACKAGE_MANAGER_ENV_VARS)
312        return dict(
313            os.environ, LANG='C', DEBIAN_FRONTEND='noninteractive'
314        )
315
316    def _write_runtime_config(self, system_default: bool = False) -> None:
317        parameters = {
318            'apt_shared_base': self.manager_base,
319            'unauthenticated': self.unauthenticated
320        }
321        if not system_default:
322            template = self.apt_conf.get_host_template(self.exclude_docs)
323            apt_conf_data = template.substitute(parameters)
324        else:
325            template = self.apt_conf.get_image_template(self.exclude_docs)
326            apt_conf_data = template.substitute(parameters)
327
328        with open(self.runtime_apt_get_config_file.name, 'w') as config:
329            config.write(apt_conf_data)
330
331    def cleanup(self) -> None:
332        """
333        Delete intermediate apt config file
334        """
335        if os.path.isfile(self.runtime_apt_get_config_file.name):
336            os.unlink(self.runtime_apt_get_config_file.name)

139    def add_repo(
140        self, name: str, uri: str, repo_type: str = 'deb',
141        prio: int = None, dist: str = None, components: str = None,
142        user: str = None, secret: str = None, credentials_file: str = None,
143        repo_gpgcheck: bool = None, pkg_gpgcheck: bool = None,
144        sourcetype: str = None, customization_script: str = None,
145        architectures: str = None
146    ) -> None:
147        """
148        Add apt_get repository
149
150        :param str name: repository base file name
151        :param str uri: repository URI
152        :param str repo_type: unused
153        :param int prio: unused
154        :param str dist: distribution name for non flat deb repos
155        :param str components: distribution categories
156        :param str user: unused
157        :param str secret: unused
158        :param str credentials_file: unused
159        :param bool repo_gpgcheck: enable repository signature validation
160        :param bool pkg_gpgcheck: unused
161        :param str sourcetype: unused
162        :param str customization_script:
163            custom script called after the repo file was created
164        :param str architectures:
165            identifies which architectures are supported by this repository
166        """
167        sources_file = '/'.join(
168            [self.shared_apt_get_dir['sources-dir'], name + '.sources']
169        )
170        pref_file = '/'.join(
171            [self.shared_apt_get_dir['preferences-dir'], name + '.pref']
172        )
173        self.repo_names.append(name + '.sources')
174        if os.path.exists(uri):
175            # apt-get requires local paths to take the file: type
176            uri = 'file:/' + uri
177            uri = uri.replace('file://', 'file:/')
178        if not components:
179            components = 'main'
180        self._add_components(components)
181        with open(sources_file, 'w') as repo:
182            repo_details = 'Types: deb' + os.linesep
183            repo_details += 'URIs: ' + uri + os.linesep
184            if architectures:
185                repo_details += 'Architectures: {}{}'.format(
186                    architectures.replace(',', ' '), os.linesep
187                )
188            if not dist:
189                # create a debian flat repository setup. We consider the
190                # repository metadata to exist on the toplevel of the
191                # specified uri. This applies to the way the open build
192                # service creates debian repositories and should be
193                # done in the same way for other repositories when used
194                # with kiwi
195                repo_details += 'Suites: ./' + os.linesep
196            else:
197                # create a debian distributon repository setup for the
198                # specified distributon name and components
199                self.distribution = dist
200                self.distribution_path = uri
201                repo_details += 'Suites: ' + dist + os.linesep
202                repo_details += 'Components: ' + components + os.linesep
203            if repo_gpgcheck is False:
204                repo_details += 'trusted: yes' + os.linesep
205                repo_details += 'check-valid-until: no' + os.linesep
206            repo.write(repo_details)
207        if customization_script:
208            self.run_repo_customize(customization_script, sources_file)
209        if prio:
210            uri_parsed = urlparse(uri.replace('file://', 'file:/'))
211            with open(pref_file, 'w') as pref:
212                pref.write('Package: *{0}'.format(os.linesep))
213                if not uri_parsed.hostname:
214                    pref.write(
215                        'Pin: origin ""{0}'.format(os.linesep)
216                    )
217                else:
218                    pref.write(
219                        'Pin: origin "{0}"{1}'.format(
220                            uri_parsed.hostname, os.linesep
221                        )
222                    )
223                pref.write(
224                    'Pin-Priority: {0}{1}'.format(prio, os.linesep)
225                )
226            if customization_script:
227                self.run_repo_customize(customization_script, pref_file)
